for(int nDelay = -nNumSamp + 1; nDelay < nNumSamp; nDelay++)
{
    float fSum = 0;

    for(int j = 0; j < nNumSamp; j++)
    {
        int kIndex = nDelay + j;

        if(kIndex < 0 || kIndex >= nNumSamp)
        {
            continue;
        }

        fSum += pfIn0[j] * pfIn1[kIndex];
    }

    pfOut[nDelay + nNumSamp] = fSum;
}
